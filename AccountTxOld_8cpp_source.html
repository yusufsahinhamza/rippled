37 // account: account,
38 // ledger_index_min: ledger_index,
39 // ledger_index_max: ledger_index,
40 // binary: boolean, // optional, defaults to false

43 // offset: integer, // optional, defaults to 0

45 // }
46 Json::Value doAccountTxOld (RPC::JsonContext& context)
47 {
48  std::uint32_t offset
49  = context.params.isMember (jss::offset)
50  ? context.params[jss::offset].asUInt () : 0;
51  int limit = context.params.isMember (jss::limit)
52  ? context.params[jss::limit].asUInt () : -1;
53  bool bBinary = context.params.isMember (jss::binary)
54  && context.params[jss::binary].asBool ();
55  bool bDescending = context.params.isMember (jss::descending)
56  && context.params[jss::descending].asBool ();
57  bool bCount = context.params.isMember (jss::count)
58  && context.params[jss::count].asBool ();
59  std::uint32_t uLedgerMin;
60  std::uint32_t uLedgerMax;
61  std::uint32_t uValidatedMin;
62  std::uint32_t uValidatedMax;
63  bool bValidated = context.ledgerMaster.getValidatedRange (
64  uValidatedMin, uValidatedMax);
65 
66  if (!context.params.isMember (jss::account))
67  return rpcError (rpcINVALID_PARAMS);
68 
69  auto const raAccount = parseBase58<AccountID>(
70  context.params[jss::account].asString());
71  if (! raAccount)
72  return rpcError (rpcACT_MALFORMED);
73 
74  if (offset > 3000)
75  return rpcError (rpcATX_DEPRECATED);
76 
77  context.loadType = Resource::feeHighBurdenRPC;
78 
79  // DEPRECATED
80  if (context.params.isMember (jss::ledger_min))
81  {
82  context.params[jss::ledger_index_min] = context.params[jss::ledger_min];
83  bDescending = true;
84  }
85 
86  // DEPRECATED
87  if (context.params.isMember (jss::ledger_max))
88  {
89  context.params[jss::ledger_index_max] = context.params[jss::ledger_max];
90  bDescending = true;
91  }
92 
93  if (context.params.isMember (jss::ledger_index_min)
94  || context.params.isMember (jss::ledger_index_max))
95  {
96  std::int64_t iLedgerMin = context.params.isMember (jss::ledger_index_min)
97  ? context.params[jss::ledger_index_min].asInt () : -1;
98  std::int64_t iLedgerMax = context.params.isMember (jss::ledger_index_max)
99  ? context.params[jss::ledger_index_max].asInt () : -1;
100 
101  if (!bValidated && (iLedgerMin == -1 || iLedgerMax == -1))
102  {
103  // Don't have a validated ledger range.
104  return rpcError (rpcLGR_IDXS_INVALID);
105  }
106 
107  uLedgerMin = iLedgerMin == -1 ? uValidatedMin : iLedgerMin;
108  uLedgerMax = iLedgerMax == -1 ? uValidatedMax : iLedgerMax;
109 
110  if (uLedgerMax < uLedgerMin)
111  {
112  return rpcError (rpcLGR_IDXS_INVALID);
113  }
114  }
115  else
116  {
117  std::shared_ptr<ReadView const> ledger;
118  auto ret = RPC::lookupLedger (ledger, context);
119 
120  if (!ledger)
121  return ret;
122 
123  if (! ret[jss::validated].asBool() ||
124  (ledger->info().seq > uValidatedMax) ||
125  (ledger->info().seq < uValidatedMin))
126  {
127  return rpcError (rpcLGR_NOT_VALIDATED);
128  }
129 
130  uLedgerMin = uLedgerMax = ledger->info().seq;
131  }
132 
133  int count = 0;
134 
135 #ifndef DEBUG
136 
137  try
138  {
139 #endif
140 
141  Json::Value ret (Json::objectValue);
142 
143  ret[jss::account] = context.app.accountIDCache().toBase58(*raAccount);
144  Json::Value& jvTxns = (ret[jss::transactions] = Json::arrayValue);
145 
146  if (bBinary)
147  {
148  auto txns = context.netOps.getAccountTxsB (
149  *raAccount, uLedgerMin, uLedgerMax, bDescending, offset, limit,
150  isUnlimited (context.role));
151 
152  for (auto it = txns.begin (), end = txns.end (); it != end; ++it)
153  {
154  ++count;
155  Json::Value& jvObj = jvTxns.append (Json::objectValue);
156 
157  std::uint32_t uLedgerIndex = std::get<2> (*it);
158  jvObj[jss::tx_blob] = strHex(std::get<0> (*it));
159  jvObj[jss::meta] = strHex(std::get<1> (*it));
160  jvObj[jss::ledger_index] = uLedgerIndex;
161  jvObj[jss::validated]
162  = bValidated
163  && uValidatedMin <= uLedgerIndex
164  && uValidatedMax >= uLedgerIndex;
165 
166  }
167  }
168  else
169  {
170  auto txns = context.netOps.getAccountTxs (
171  *raAccount, uLedgerMin, uLedgerMax, bDescending, offset, limit,
172  isUnlimited (context.role));
173 
174  for (auto it = txns.begin (), end = txns.end (); it != end; ++it)
175  {
176  ++count;
177  Json::Value& jvObj = jvTxns.append (Json::objectValue);
178 
179  if (it->first)
180  jvObj[jss::tx] =
181  it->first->getJson (JsonOptions::include_date);
182 
183  if (it->second)
184  {
185  std::uint32_t uLedgerIndex = it->second->getLgrSeq ();
186 
187  auto meta = it->second->getJson(JsonOptions::none);
188  insertDeliveredAmount(meta, context, it->first, *it->second);
189  jvObj[jss::meta] = std::move(meta);
190 
191  jvObj[jss::validated]
192  = bValidated
193  && uValidatedMin <= uLedgerIndex
194  && uValidatedMax >= uLedgerIndex;
195  }
196 
197  }
198  }
199 
200  //Add information about the original query
201  ret[jss::ledger_index_min] = uLedgerMin;
202  ret[jss::ledger_index_max] = uLedgerMax;
203  ret[jss::validated]
204  = bValidated
205  && uValidatedMin <= uLedgerMin
206  && uValidatedMax >= uLedgerMax;
207  ret[jss::offset] = offset;
208 
209  // We no longer return the full count but only the count of returned
210  // transactions. Computing this count was two expensive and this API is
211  // deprecated anyway.
212  if (bCount)
213  ret[jss::count] = count;
214 
215  if (context.params.isMember (jss::limit))
216  ret[jss::limit] = limit;
217 
218 
219  return ret;
220 #ifndef DEBUG
221  }
222  catch (std::exception const&)
223  {
224  return rpcError (rpcINTERNAL);
225  }
226 
227 #endif
228 }
