50 template <class FwdIter>
51 float
52 distribution_factor (FwdIter first, FwdIter last)
53 {
54  using value_type = typename FwdIter::value_type;
55  static_assert (std::is_unsigned <value_type>::value, "");
56 
57  const unsigned nbits = CHAR_BIT * sizeof(std::size_t);
58  const unsigned rows = nbits / 4;
59  unsigned counts[rows][16] = {};
60  std::for_each (first, last, [&](typename FwdIter::value_type h)
61  {
62  std::size_t mask = 0xF;
63  for (unsigned i = 0; i < rows; ++i, mask <<= 4)
64  counts[i][(h & mask) >> 4*i] += 1;
65  });
66  float mean_rows[rows] = {0};
67  float mean_cols[16] = {0};
68  for (unsigned i = 0; i < rows; ++i)
69  {
70  for (unsigned j = 0; j < 16; ++j)
71  {
72  mean_rows[i] += counts[i][j];
73  mean_cols[j] += counts[i][j];
74  }
75  }
76  for (unsigned i = 0; i < rows; ++i)
77  mean_rows[i] /= 16;
78  for (unsigned j = 0; j < 16; ++j)
79  mean_cols[j] /= rows;
80  std::pair<float, float> dev[rows][16];
81  for (unsigned i = 0; i < rows; ++i)
82  {
83  for (unsigned j = 0; j < 16; ++j)
84  {
85  dev[i][j].first = std::abs(counts[i][j] - mean_rows[i]) / mean_rows[i];
86  dev[i][j].second = std::abs(counts[i][j] - mean_cols[j]) / mean_cols[j];
87  }
88  }
89  float max_err = 0;
90  for (unsigned i = 0; i < rows; ++i)
91  {
92  for (unsigned j = 0; j < 16; ++j)
93  {
94  if (max_err < dev[i][j].first)
95  max_err = dev[i][j].first;
96  if (max_err < dev[i][j].second)
97  max_err = dev[i][j].second;
98  }
99  }
100  return max_err;
101 }

115 double
116 score (int const* bins, std::size_t const bincount, double const k)
117 {
118  double const n = bincount;
119  // compute rms^2 value
120  double rms_sq = 0;
121  for(std::size_t i = 0; i < bincount; ++i)
122  rms_sq += sqr(bins[i]);;
123  rms_sq /= n;
124  // compute fill factor
125  double const f = (sqr(k) - 1) / (n*rms_sq - k);
126  // rescale to (0,1) with 0 = good, 1 = bad
127  return 1 - (f / n);
128 }

130 template <class T>
131 std::uint32_t
132 window (T* blob, int start, int count )
133 {
134  std::size_t const len = sizeof(T);
135  static_assert((len & 3) == 0, "");
136  if(count == 0)
137  return 0;
138  int const nbits = len * CHAR_BIT;
139  start %= nbits;
140  int ndwords = len / 4;
141  std::uint32_t const* k = static_cast <
142  std::uint32_t const*>(static_cast<void const*>(blob));
143  int c = start & (32-1);
144  int d = start / 32;
145  if(c == 0)
146  return (k[d] & ((1 << count) - 1));
147  int ia = (d + 1) % ndwords;
148  int ib = (d + 0) % ndwords;
149  std::uint32_t a = k[ia];
150  std::uint32_t b = k[ib];
151  std::uint32_t t = (a << (32-c)) | (b >> c);
152  t &= ((1 << count)-1);
153  return t;
154 }

161 template <class FwdIter>
162 double
163 windowed_score (FwdIter first, FwdIter last)
164 {
165  auto const size (std::distance (first, last));
166  int maxwidth = 20;
167  // We need at least 5 keys per bin to reliably test distribution biases
168  // down to 1%, so don't bother to test sparser distributions than that
169  while (static_cast<double>(size) / (1ull << maxwidth) < 5.0)
170  maxwidth--;
171  double worst = 0;
172  std::vector <int> bins (1ull << maxwidth);
173  int const hashbits = sizeof(std::size_t) * CHAR_BIT;
174  for (int start = 0; start < hashbits; ++start)
175  {
176  int width = maxwidth;
177  bins.assign (1ull << width, 0);
178  for (auto iter (first); iter != last; ++iter)
179  ++bins[detail::window(&*iter, start, width)];
180  // Test the distribution, then fold the bins in half,
181  // repeat until we're down to 256 bins
182  while (bins.size() >= 256)
183  {
184  double score (detail::score (
185  bins.data(), bins.size(), size));
186  worst = std::max(score, worst);
187  if (--width < 8)
188  break;
189  for (std::size_t i = 0, j = bins.size() / 2; j < bins.size(); ++i, ++j)
190  bins[i] += bins[j];
191  bins.resize(bins.size() / 2);
192  }
193  }
194  return worst;
195 }
