40  explicit Impl(
41  DigestAwareReadView const& ledger,
42  std::unordered_set<uint256, beast::uhash<>> const& presets)
43  : presets_(presets)
44  {
45  auto const k = keylet::amendments();
46  digest_ = ledger.digest(k.key);
47  if (! digest_)
48  return;
49  auto const sle = ledger.read(k);
50  if (! sle)
51  {
52  // LogicError() ?
53  return;
54  }
55 
56  for (auto const& item :
57  sle->getFieldV256(sfAmendments))
58  set_.insert(item);
59  }
60 
61  bool
62  enabled (uint256 const& feature) const
63  {
64  if (presets_.count(feature) > 0)
65  return true;
66  return set_.count(feature) > 0;
67  }
68 
69  bool
70  changed (DigestAwareReadView const& ledger) const
71  {
72  auto const digest =
73  ledger.digest(keylet::amendments().key);
74  if (! digest && ! digest_)
75  return false;
76  if (! digest || ! digest_)
77  return true;
78  return *digest != *digest_;
79  }
80 
81  bool
82  operator== (Impl const& other) const
83  {
84  if (! digest_ && ! other.digest_)
85  return true;
86  if (! digest_ || ! other.digest_)
87  return false;
88  return *digest_ == *other.digest_;
89  }
