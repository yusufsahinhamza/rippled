26 // endian provides answers to the following questions:
27 // 1. Is this system big or little endian?
28 // 2. Is the "desired endian" of some class or function the same as the
29 // native endian?
30 enum class endian {
31 #ifdef _MSC_VER
32  big = 1,
33  little = 0,
34  native = little
35 #else
36  native = __BYTE_ORDER__,
37  little = __ORDER_LITTLE_ENDIAN__,
38  big = __ORDER_BIG_ENDIAN__
39 #endif
40 };
