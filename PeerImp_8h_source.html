658 template <class FwdIt, class>
659 void
660 PeerImp::sendEndpoints(FwdIt first, FwdIt last)
661 {
662  protocol::TMEndpoints tm;
663  for (; first != last; ++first)
664  {
665  auto const& ep = *first;
666  // eventually remove endpoints and just keep endpoints_v2
667  // (once we are sure the entire network understands endpoints_v2)
668  protocol::TMEndpoint& tme(*tm.add_endpoints());
669  if (ep.address.is_v4())
670  tme.mutable_ipv4()->set_ipv4(boost::endian::native_to_big(
671  static_cast<std::uint32_t>(ep.address.to_v4().to_ulong())));
672  else
673  tme.mutable_ipv4()->set_ipv4(0);
674  tme.mutable_ipv4()->set_ipv4port(ep.address.port());
675  tme.set_hops(ep.hops);
676 
677  // add v2 endpoints (strings)
678  auto& tme2(*tm.add_endpoints_v2());
679  tme2.set_endpoint(ep.address.to_string());
680  tme2.set_hops(ep.hops);
681  }
682  tm.set_version(2);
683 
684  send(std::make_shared<Message>(tm, protocol::mtENDPOINTS));
685 }
