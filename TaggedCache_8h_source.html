158  void sweep ()
159  {
160  int cacheRemovals = 0;
161  int mapRemovals = 0;
162  int cc = 0;
163 
164  // Keep references to all the stuff we sweep
165  // so that we can destroy them outside the lock.
166  //
167  std::vector <mapped_ptr> stuffToSweep;
168 
169  {
170  clock_type::time_point const now (m_clock.now());
171  clock_type::time_point when_expire;
172 
173  std::lock_guard lock (m_mutex);
174 
175  if (m_target_size == 0 ||
176  (static_cast<int> (m_cache.size ()) <= m_target_size))
177  {
178  when_expire = now - m_target_age;
179  }
180  else
181  {
182  when_expire = now - m_target_age*m_target_size/m_cache.size();
183 
184  clock_type::duration const minimumAge (
185  std::chrono::seconds (1));
186  if (when_expire > (now - minimumAge))
187  when_expire = now - minimumAge;
188 
189  JLOG(m_journal.trace()) <<
190  m_name << " is growing fast " << m_cache.size () << " of " << m_target_size <<
191  " aging at " << (now - when_expire).count() << " of " << m_target_age.count();
192  }
193 
194  stuffToSweep.reserve (m_cache.size ());
195 
196  cache_iterator cit = m_cache.begin ();
197 
198  while (cit != m_cache.end ())
199  {
200  if (cit->second.isWeak ())
201  {
202  // weak
203  if (cit->second.isExpired ())
204  {
205  ++mapRemovals;
206  cit = m_cache.erase (cit);
207  }
208  else
209  {
210  ++cit;
211  }
212  }
213  else if (cit->second.last_access <= when_expire)
214  {
215  // strong, expired
216  --m_cache_count;
217  ++cacheRemovals;
218  if (cit->second.ptr.unique ())
219  {
220  stuffToSweep.push_back (cit->second.ptr);
221  ++mapRemovals;
222  cit = m_cache.erase (cit);
223  }
224  else
225  {
226  // remains weakly cached
227  cit->second.ptr.reset ();
228  ++cit;
229  }
230  }
231  else
232  {
233  // strong, not expired
234  ++cc;
235  ++cit;
236  }
237  }
238  }
239 
240  if (mapRemovals || cacheRemovals)
241  {
242  JLOG(m_journal.trace()) <<
243  m_name << ": cache = " << m_cache.size () <<
244  "-" << cacheRemovals << ", map-=" << mapRemovals;
245  }
246 
247  // At this point stuffToSweep will go out of scope outside the lock
248  // and decrement the reference count on each strong pointer.
249  }

251  bool del (const key_type& key, bool valid)
252  {
253  // Remove from cache, if !valid, remove from map too. Returns true if removed from cache
254  std::lock_guard lock (m_mutex);
255 
256  cache_iterator cit = m_cache.find (key);
257 
258  if (cit == m_cache.end ())
259  return false;
260 
261  Entry& entry = cit->second;
262 
263  bool ret = false;
264 
265  if (entry.isCached ())
266  {
267  --m_cache_count;
268  entry.ptr.reset ();
269  ret = true;
270  }
271 
272  if (!valid || entry.isExpired ())
273  m_cache.erase (cit);
274 
275  return ret;
276  }

293  template <bool replace>
294  bool
295  canonicalize(
296  const key_type& key,
297  std::conditional_t<replace, std::shared_ptr<T> const, std::shared_ptr<T>>& data
298  )
299  {
300  // Return canonical value, store if needed, refresh in cache
301  // Return values: true=we had the data already
302  std::lock_guard lock (m_mutex);
303 
304  cache_iterator cit = m_cache.find (key);
305 
306  if (cit == m_cache.end ())
307  {
308  m_cache.emplace (std::piecewise_construct,
309  std::forward_as_tuple(key),
310  std::forward_as_tuple(m_clock.now(), data));
311  ++m_cache_count;
312  return false;
313  }
314 
315  Entry& entry = cit->second;
316  entry.touch (m_clock.now());
317 
318  if (entry.isCached ())
319  {
320  if constexpr (replace)
321  {
322  entry.ptr = data;
323  entry.weak_ptr = data;
324  }
325  else
326  {
327  data = entry.ptr;
328  }
329 
330  return true;
331  }
332 
333  mapped_ptr cachedData = entry.lock ();
334 
335  if (cachedData)
336  {
337  if constexpr (replace)
338  {
339  entry.ptr = data;
340  entry.weak_ptr = data;
341  }
342  else
343  {
344  entry.ptr = cachedData;
345  data = cachedData;
346  }
347 
348  ++m_cache_count;
349  return true;
350  }
351 
352  entry.ptr = data;
353  entry.weak_ptr = data;
354  ++m_cache_count;
355 
356  return false;
357  }

370  std::shared_ptr<T> fetch (const key_type& key)
371  {
372  // fetch us a shared pointer to the stored data object
373  std::lock_guard lock (m_mutex);
374 
375  cache_iterator cit = m_cache.find (key);
376 
377  if (cit == m_cache.end ())
378  {
379  ++m_misses;
380  return mapped_ptr ();
381  }
382 
383  Entry& entry = cit->second;
384  entry.touch (m_clock.now());
385 
386  if (entry.isCached ())
387  {
388  ++m_hits;
389  return entry.ptr;
390  }
391 
392  entry.ptr = entry.lock ();
393 
394  if (entry.isCached ())
395  {
396  // independent of cache size, so not counted as a hit
397  ++m_cache_count;
398  return entry.ptr;
399  }
400 
401  m_cache.erase (cit);
402  ++m_misses;
403  return mapped_ptr ();
404  }

439  bool refreshIfPresent (const key_type& key)
440  {
441  bool found = false;
442 
443  // If present, make current in cache
444  std::lock_guard lock (m_mutex);
445 
446  cache_iterator cit = m_cache.find (key);
447 
448  if (cit != m_cache.end ())
449  {
450  Entry& entry = cit->second;
451 
452  if (! entry.isCached ())
453  {
454  // Convert weak to strong.
455  entry.ptr = entry.lock ();
456 
457  if (entry.isCached ())
458  {
459  // We just put the object back in cache
460  ++m_cache_count;
461  entry.touch (m_clock.now());
462  found = true;
463  }
464  else
465  {
466  // Couldn't get strong pointer,
467  // object fell out of the cache so remove the entry.
468  m_cache.erase (cit);
469  }
470  }
471  else
472  {
473  // It's cached so update the timer
474  entry.touch (m_clock.now());
475  found = true;
476  }
477  }
478  else
479  {
480  // not present
481  }
482 
483  return found;
484  }
