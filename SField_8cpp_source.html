280 int SField::compare (SField const& f1, SField const& f2)
281 {
282  // -1 = f1 comes before f2, 0 = illegal combination, 1 = f1 comes after f2
283  if ((f1.fieldCode <= 0) || (f2.fieldCode <= 0))
284  return 0;
285 
286  if (f1.fieldCode < f2.fieldCode)
287  return -1;
288 
289  if (f2.fieldCode < f1.fieldCode)
290  return 1;
291 
292  return 0;
293 }
