64 template <std::size_t Bits, class Tag = void>
65 class base_uint

299  base_uint& operator+= (const base_uint& b)
300  {
301  std::uint64_t carry = 0;
302 
303  for (int i = WIDTH; i--;)
304  {
305  std::uint64_t n = carry + boost::endian::big_to_native(data_[i]) +
306  boost::endian::big_to_native(b.data_[i]);
307 
308  data_[i] = boost::endian::native_to_big (static_cast<std::uint32_t>(n));
309  carry = n >> 32;
310  }
311 
312  return *this;
313  }

327  bool SetHexExact (const char* psz)
328  {
329  unsigned char* pOut = begin ();
330 
331  for (int i = 0; i < sizeof (data_); ++i)
332  {
333  auto hi = charUnHex(*psz++);
334  if (hi == -1)
335  return false;
336 
337  auto lo = charUnHex (*psz++);
338  if (lo == -1)
339  return false;
340 
341  *pOut++ = (hi << 4) | lo;
342  }
343 
344  // We've consumed exactly as many bytes as we needed at this point
345  // so we should be at the end of the string.
346  return (*psz == 0);
347  }
348 
362  bool SetHex (const char* psz, bool bStrict = false)
363  {
364  // Find beginning.
365  auto pBegin = reinterpret_cast<const unsigned char*>(psz);
366  // skip leading spaces
367  if (!bStrict)
368  while (isspace(*pBegin))
369  pBegin++;
370 
371  // skip 0x
372  if (!bStrict && pBegin[0] == '0' && tolower(pBegin[1]) == 'x')
373  pBegin += 2;
374 
375  // Find end.
376  auto pEnd = pBegin;
377  while (charUnHex(*pEnd) != -1)
378  pEnd++;
379 
380  // Take only last digits of over long string.
381  if ((unsigned int) (pEnd - pBegin) > 2 * size ())
382  pBegin = pEnd - 2 * size ();
383 
384  unsigned char* pOut = end () - ((pEnd - pBegin + 1) / 2);
385 
386  *this = beast::zero;
387 
388  if ((pEnd - pBegin) & 1)
389  *pOut++ = charUnHex(*pBegin++);
390 
391  while (pBegin != pEnd)
392  {
393  auto cHigh = charUnHex(*pBegin++);
394  auto cLow = pBegin == pEnd
395  ? 0
396  : charUnHex(*pBegin++);
397 
398  if (cHigh == -1 || cLow == -1)
399  return false;
400 
401  *pOut++ = (cHigh << 4) | cLow;
402  }
403 
404  return !*pEnd;
405  }

438 template <std::size_t Bits, class Tag>
439 inline int compare (
440  base_uint<Bits, Tag> const& a, base_uint<Bits, Tag> const& b)
441 {
442  auto ret = std::mismatch (a.cbegin (), a.cend (), b.cbegin ());
443 
444  if (ret.first == a.cend ())
445  return 0;
446 
447  // a > b
448  if (*ret.first > *ret.second)
449  return 1;
450 
451  // a < b
452  return -1;
453 }
