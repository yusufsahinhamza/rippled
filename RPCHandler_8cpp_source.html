114 error_code_i fillHandler (JsonContext& context,
115  Handler const * & result)
116 {
117  if (! isUnlimited (context.role))
118  {
119  // VFALCO NOTE Should we also add up the jtRPC jobs?
120  //
121  int jc = context.app.getJobQueue ().getJobCountGE (jtCLIENT);
122  if (jc > Tuning::maxJobQueueClients)
123  {
124  JLOG (context.j.debug()) << "Too busy for command: " << jc;
125  return rpcTOO_BUSY;
126  }
127  }
128 
129  if (!context.params.isMember(jss::command) && !context.params.isMember(jss::method))
130  return rpcCOMMAND_MISSING;
131  if (context.params.isMember(jss::command) && context.params.isMember(jss::method))
132  {
133  if (context.params[jss::command].asString() !=
134  context.params[jss::method].asString())
135  return rpcUNKNOWN_COMMAND;
136  }
137 
138  std::string strCommand = context.params.isMember(jss::command) ?
139  context.params[jss::command].asString() :
140  context.params[jss::method].asString();
141 
142  JLOG (context.j.trace()) << "COMMAND:" << strCommand;
143  JLOG (context.j.trace()) << "REQUEST:" << context.params;
144  auto handler = getHandler(context.apiVersion, strCommand);
145 
146  if (!handler)
147  return rpcUNKNOWN_COMMAND;
148 
149  if (handler->role_ == Role::ADMIN && context.role != Role::ADMIN)
150  return rpcNO_PERMISSION;
151 
152  error_code_i res = conditionMet(handler->condition_,context);
153  if(res != rpcSUCCESS)
154  {
155  return res;
156  }
157 
158  result = handler;
159  return rpcSUCCESS;
160 }

162 template <class Object, class Method>
163 Status callMethod (
164  JsonContext& context, Method method, std::string const& name, Object& result)
165 {
166  static std::atomic<std::uint64_t> requestId {0};
167  auto& perfLog = context.app.getPerfLog();
168  std::uint64_t const curId = ++requestId;
169  try
170  {
171  perfLog.rpcStart(name, curId);
172  auto v = context.app.getJobQueue().makeLoadEvent(
173  jtGENERIC, "cmd:" + name);
174 
175  auto ret = method (context, result);
176  perfLog.rpcFinish(name, curId);
177  return ret;
178  }
179  catch (std::exception& e)
180  {
181  perfLog.rpcError(name, curId);
182  JLOG (context.j.info()) << "Caught throw: " << e.what ();
183 
184  if (context.loadType == Resource::feeReferenceRPC)
185  context.loadType = Resource::feeExceptionRPC;
186 
187  inject_error (rpcINTERNAL, result);
188  return rpcINTERNAL;
189  }
190 }

194 Status doCommand (
195  RPC::JsonContext& context, Json::Value& result)
196 {
197  Handler const * handler = nullptr;
198  if (auto error = fillHandler (context, handler))
199  {
200  inject_error (error, result);
201  return error;
202  }
203 
204  if (auto method = handler->valueMethod_)
205  {
206  if (! context.headers.user.empty() ||
207  ! context.headers.forwardedFor.empty())
208  {
209  JLOG(context.j.debug()) << "start command: " << handler->name_ <<
210  ", user: " << context.headers.user << ", forwarded for: " <<
211  context.headers.forwardedFor;
212 
213  auto ret = callMethod (context, method, handler->name_, result);
214 
215  JLOG(context.j.debug()) << "finish command: " << handler->name_ <<
216  ", user: " << context.headers.user << ", forwarded for: " <<
217  context.headers.forwardedFor;
218 
219  return ret;
220  }
221  else
222  {
223  return callMethod (context, method, handler->name_, result);
224  }
225  }
226 
227  return rpcUNKNOWN_COMMAND;
228 }
