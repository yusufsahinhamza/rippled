312 struct LedgerHistoryHelper
313 {
314  LedgerOracle oracle;
315  Tx::ID nextTx{0};
316  std::unordered_map<std::string, Ledger> ledgers;
317  std::set<char> seen;
318 
319  LedgerHistoryHelper()
320  {
321  ledgers[""] = Ledger{Ledger::MakeGenesis{}};
322  }
323 
329  Ledger const& operator[](std::string const& s)
330  {
331  auto it = ledgers.find(s);
332  if (it != ledgers.end())
333  return it->second;
334 
335  // enforce that the new suffix has never been seen
336  assert(seen.emplace(s.back()).second);
337 
338  Ledger const& parent = (*this)[s.substr(0, s.size() - 1)];
339  return ledgers.emplace(s, oracle.accept(parent, ++nextTx))
340  .first->second;
341  }
