43 void Config::applyTuning ()
44 {
45  if (maxPeers < Tuning::minOutCount)
46  maxPeers = Tuning::minOutCount;
47  outPeers = calcOutPeers ();
48 
49  auto const inPeers = maxPeers - outPeers;
50 
51  if (ipLimit == 0)
52  {
53  // Unless a limit is explicitly set, we allow between
54  // 2 and 5 connections from non RFC-1918 "private"
55  // IP addresses.
56  ipLimit = 2;
57 
58  if (inPeers > Tuning::defaultMaxPeers)
59  ipLimit += std::min(5,
60  static_cast<int>(inPeers / Tuning::defaultMaxPeers));
61  }
62 
63  // We don't allow a single IP to consume all incoming slots,
64  // unless we only have one incoming slot available.
65  ipLimit = std::max(1,
66  std::min(ipLimit, static_cast<int>(inPeers / 2)));
67 }
